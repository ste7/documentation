---
title: "Get Current User"
openapi: "GET /v1/users/me"
---

Retrieve information about the currently authenticated user.

## Response

<ResponseField name="success" type="boolean" required>
  Indicates whether the request was successful. Always `true` for successful
  responses.
</ResponseField>

<ResponseField name="data" type="object">
  The user object.
  
  <Expandable title="User properties">
    <ResponseField name="id" type="integer" required>
      The unique identifier of the user.
    </ResponseField>

    <ResponseField name="first_name" type="string" required>
      The user's first name.
    </ResponseField>

    <ResponseField name="last_name" type="string" required>
      The user's last name.
    </ResponseField>

    <ResponseField name="username" type="string" required>
      The unique username for the user.
    </ResponseField>

    <ResponseField name="email" type="string" required>
      The user's email address.
    </ResponseField>

    <ResponseField name="timezone" type="string" required>
      The user's timezone (e.g., "America/New_York").
    </ResponseField>

    <ResponseField name="time_format" type="string" required>
      The user's preferred time format (e.g., "12h" or "24h").
    </ResponseField>

    <ResponseField name="start_of_week" type="string" required>
      The day that starts the week (e.g., "monday", "sunday").
    </ResponseField>

    <ResponseField name="avatar" type="string">
      URL to the user's avatar image. Returns `null` if no avatar is set.
    </ResponseField>

  </Expandable>
</ResponseField>
