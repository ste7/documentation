---
title: "Get Event Types"
openapi: "GET /v1/event-types"
---

Retrieve all event types that the authenticated user has access to. This endpoint returns a list of event types owned by the user or accessible through team memberships.

## Response

<ResponseField name="success" type="boolean" required>
  Indicates whether the request was successful. Always `true` for successful
  responses.
</ResponseField>

<ResponseField name="data" type="array">
  Array of event type objects.
  
  <Expandable title="Event Type properties">
    <ResponseField name="id" type="integer" required>
      The unique identifier of the event type.
    </ResponseField>

    <ResponseField name="name" type="string" required>
      The name of the event type (max 55 characters).
    </ResponseField>

    <ResponseField name="slug" type="string" required>
      The URL-friendly identifier (max 80 characters, lowercase alphanumeric with hyphens).
    </ResponseField>

    <ResponseField name="description" type="string">
      Description of the event type (max 5000 characters).
    </ResponseField>

    <ResponseField name="type" type="string" required>
      The type of event. Possible values: `one_on_one`, `group`, `round_robin`, `collective`.
    </ResponseField>

    <ResponseField name="duration" type="integer" required>
      Duration in minutes (1-1440).
    </ResponseField>

    <ResponseField name="max_group_size" type="integer">
      Maximum group size for group events (2-10). Required when `type` is `group`.
    </ResponseField>

    <ResponseField name="hidden_from_profile" type="boolean" required>
      Whether the event type is hidden from the user's public profile.
    </ResponseField>

    <ResponseField name="color" type="string" required>
      Color code in hex format (e.g., `#FF5733`).
    </ResponseField>

    <ResponseField name="form_fields" type="array">
      Array of active form fields for this event type.

      <Expandable title="Form Field properties">
        <ResponseField name="identifier" type="string" required>
          Unique identifier for the form field.
        </ResponseField>

        <ResponseField name="label" type="string" required>
          Display label for the form field.
        </ResponseField>

        <ResponseField name="type" type="string" required>
          The type of form field (e.g., `text`, `email`, `select`).
        </ResponseField>

        <ResponseField name="placeholder" type="string">
          Placeholder text for the form field.
        </ResponseField>

        <ResponseField name="is_required" type="boolean" required>
          Whether the form field is required.
        </ResponseField>

        <ResponseField name="options" type="array">
          Array of options for select/dropdown fields.
        </ResponseField>
      </Expandable>
    </ResponseField>

  </Expandable>
</ResponseField>
